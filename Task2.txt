//Выведите на экран 10 раз фразу "You are welcome!" (через for, while, do while)
for (var i = 0; i<10; i++){
console.log(" You are welcome! ");
}
//while
var i = 0;
while (i<10 ){
console.log(" You are welcome! " + "\n");
i++;
}
//do while
var i = 0;
do {
  console.log( "You are welcome!" + "\n" );
  i++;
} while (i < 10);
